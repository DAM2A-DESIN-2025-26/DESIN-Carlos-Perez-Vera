<!-- Campo título -->
<div>
  <!-- Etiqueta asociada al input mediante el id -->
  <label for="titulo">Título</label>

  <!--
    [value]="titulo()" :
    muestra en el input el valor actual de la signal titulo

    (input)="titulo.set(...)" :
    cada vez que el usuario escribe, actualizamos la signal titulo
    con el valor del input
  -->
  <input
    id="titulo"
    type="text"
    [value]="titulo()"
    (input)="titulo.set($any($event.target).value)"
  />
</div>

<div>

  <textarea
    id="descripcion"
    [value]="descripcion()"
    (input)="descripcion.set($any($event.target).value)"
  ></textarea>
</div>

<div>
  <label for="estado">Estado</label>

  <!--
    [value]="estado()" :
    marca como seleccionada la opción que tenga el valor actual de la signal estado

  <select

<!-- Select de usuario -->
<div>
  <label for="usuario">Usuario</label>

  <!--
    Este select guarda el id del usuario seleccionado

    OJO:
    el value de un select llega como texto, pero usuarioId es number.
    Por eso usamos + delante:
    +$any($event.target).value

    Ese + convierte el texto a número
  -->
  <select
    id="usuario"
    [value]="usuarioId()"
    (change)="usuarioId.set(+$any($event.target).value)"
  >
    <!-- Opción inicial por defecto -->
    <option value="0">Selecciona un usuario</option>

    <!--
      @for recorre el array usuarios()

      usuarios() :
      lee el valor actual de la signal usuarios

      track usuario.id :
      ayuda a Angular a identificar cada opción por su id
    -->
    @for (usuario of usuarios(); track usuario.id) {
      <option [value]="usuario.id">
        {{ usuario.nombre }}
      </option>
    }
  </select>
</div>

<!-- Select de categoría -->
<div>

  <select
    id="categoria"
    [value]="categoriaId()"
    (change)="categoriaId.set(+$any($event.target).value)"
  >
    <option value="0">Selecciona una categoría</option>

    <!--
      Recorremos la signal categorias para generar las opciones
    -->
    @for (categoria of categorias(); track categoria.id) {
      <option [value]="categoria.id">
        {{ categoria.nombre }}
      </option>
    }
  </select>
</div>
